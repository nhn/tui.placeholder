        $('#btn-remove').on('click', function() {
            var $parent = $('#add-area');
            var $removeElements = $('#add-area').find('input');

            placeholder.remove($removeElements);

            $parent.empty();
        });

        $('#btn-hide').on('click', function() {
            var $target = $('#add-area').find('input').eq(1);

            tui.placeholder.hide($target);
        });
